** Command
#+begin_src python
from clion import Clion


app = Clion()


@app.command()
def hello(name):
    """Greet a person"""
    print(f"Hello, {name}!")


if __name__ == "__main__":
    app()
#+end_src

** Actions
#+begin_src python
from clion import Clion


app = Clion()


@app.command()
def hello():
    """Greet a person"""


@hello.action()
def hello_english(name):
    """Greet a person in english"""
    print(f"Hello, {name}!")


@hello.action()
def hello_french(name):
    """Greet a person in french"""
    print(f"Salut, {name}!")


if __name__ == "__main__":
    app()
#+end_src

#+begin_example
usage: test.py hello [-h] {english,french} ...

positional arguments:
  {english,french}  action to do
    english         Greet a person in english
    french          Greet a person in french

optional arguments:
  -h, --help        show this help message and exit
#+end_example
